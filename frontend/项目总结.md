## 微信小程序集成建议

### 1. 数据导入
```javascript
// 在 app.js 中加载数据
const furnitureData = require('./data/furniture.json')
const textileData = require('./data/textile.json')
const foodData = require('./data/food.json')

App({
  globalData: {
    furniture: furnitureData,
    textile: textileData,
    food: foodData
  }
})
```

### 2. 材质识别功能
```javascript
// 根据材质名称查询风险
function getMaterialRisk(materialType) {
  const app = getApp()
  const material = app.globalData.furniture.find(
    item => item.data.material_type === materialType
  )
  return material ? material.data.risk_points : []
}
```

### 3. 风险等级展示
```javascript
// 根据风险等级显示颜色
function getRiskColor(severity) {
  const colors = {
    '高': '#ff4d4f',    // 红色
    '中': '#faad14',    // 黄色
    '低': '#52c41a'     // 绿色
  }
  return colors[severity] || '#d9d9d9'
}
```
